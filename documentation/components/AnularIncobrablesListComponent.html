import { Component, OnInit, AfterViewInit, ViewChild } from '@angular/core';
import { AnularIncobrables } from '../anular-incobrables.model';
import { Router } from '@angular/router';
import { ApiService } from 'src/app/core/services/api.service';
import { ToasterService } from 'src/app/core/services/toaster.service';
import { delay } from 'q';

import { MatTableDataSource } from '@angular/material/table';
import { TransitiveCompileNgModuleMetadata } from '@angular/compiler';


declare let $: any;

@Component({
  selector: 'app-anular-incobrables-list',
  templateUrl: './anular-incobrables-list.component.html',
  styleUrls: ['./anular-incobrables-list.component.css']
})
export class AnularIncobrablesListComponent implements OnInit, AfterViewInit {

  // retenciones: RetencionesIva[] = [];
  tableTitle: string = 'A Anular por Incobrables'
  panelExpanded: boolean = true;
  retenciones: any;
  columns: object;
  rowDetails: any[];
  filtersFields: object;
  filterModel: object;
  columnsVisibility: object;
  isLoadingResults: boolean = false;

  constructor(
    private apiService: ApiService, private router: Router, private toaster: ToasterService) {
  }

  ngOnInit() {
    // this.getRetenciones();
    this.retenciones = [];
    this.columns = {
      select : { label: 'Select', class: 'col' },
      seccion: { label: 'Seccion', visible: true, class: 'col' },
      poliza: { label: 'Poliza', visible: true, class: 'col' },
      end: { label: 'End', visible: true, class: 'col' },
      asegurado: { label: 'Asegurado', visible: true, class: 'col' },
      productor: { label: 'Productor', visible: true, class: 'col' },
      nombre: { label: 'Nombre Productor', visible: true, class: 'col' },
      agencia: { label: 'Agencia', visible: true, class: 'col' },
      fecha_emision: { label: 'Fecha Emision', visible: true, class: 'col' },
      fecha_inicio: { label: 'Fecha Inicio', visible: true, class: 'col' },
      fecha_vto: { label: 'Fecha Vto', visible: true, class: 'col' },
      moneda: { label: 'Moneda', visible: true, class: 'col' },
      premio: { label: 'Premio', visible: true, class: 'col' },
      saldo: { label: 'Saldo', visible: true, class: 'col' },
      fecha_1er_cuo: { label: 'Fecha 1er Cuo', visible: true, class: 'col' },
      descargar: { label: 'Descargar', class: 'col', action: true, icon: 'far fa-file-pdf fa-lg' }
    }

    this.filtersFields = {
      productor : {
        label: { text: 'Productor', class: 'col-sm-4' },
        control: { 
          type: 'typeahead',
          class: 'col-sm-8',
          path: '/productores',
          options: {
            value: 'nombre',
            key: 'idProductor'
          },
          pasteFieldOnSelect: 'idProductor',
          defaultValue: '',
        },
        class: 'col-sm-12 col-md-12 col-lg-4',
        required: true
      },
      seccion: {
        label: 'Seccion',
        control: { 
          type: 'select',
          path: '/seccion',
          options: {
            key: 'idSeccion',
            value: 'seccion'
          },
          pasteFieldOnSelect: 'seccion',
          multiselect: true
        },
        class: 'col-sm-12 col-md-12 col-lg-4',
        required: true
      },
      fecha_emision: {
        label: 'Exigible Hasta',
        control: { 
          type: 'datepicker',
          format: 'dd/mm/yyyy'
        },
        class: 'col-sm-12 col-md-12 col-lg-4',
        required: true
      }
    };

  }


  ngAfterViewInit() {}

  getRetenciones() {
    /*this.modelService.getAll().subscribe(
      siniestros => this.retenciones = siniestros
    );*/

  }

  applyFilters(oFilterData: object) {
    const filterString = this.apiService.generateStringFilter(oFilterData,this.filtersFields);
    console.log('query iva', filterString);
    this.isLoadingResults = true;
    this.apiService.get(`/anular-incobrables${filterString}`).subscribe( (data : object[]) => {
      console.log('anular-incobrables get', data);
      this.retenciones = new MatTableDataSource(data);
      delay(5000);
      this.isLoadingResults = false;
      if(data.length > 0){
        this.panelExpanded = false;
      }
    })
    
  }

}
