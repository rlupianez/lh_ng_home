import { Component, OnInit, Input, Output, EventEmitter, AfterViewChecked } from '@angular/core';
import { FormGroup, FormBuilder, FormControl, Validators, AbstractControl} from '@angular/forms';

import { FormsValidatorService } from '../../../core/services/forms-validator.service';

import { ChangeDetectorRef } from '@angular/core';

import * as moment from 'moment';


// generar servicio

export function RequireMatch(control: AbstractControl) {
  const selection: any = control.value;
  if (typeof selection === 'string') {
      return { incorrect: true };
  }
  return null;
}

export interface Control {
  type: string;
  class: string;
  path: string;
  options?: object
  pasteValue: string;
  defaultValue: string;
  actions: object;
}

export interface Filter {
  label: object | string;
  control: Control;
  fieldname: string;
  class: string;
  required: boolean;
}

@Component({
  selector: 'app-filters-toolbar',
  templateUrl: './filters-toolbar.component.html',
  styleUrls: ['./filters-toolbar.component.scss']
})
export class FiltersToolbarComponent implements OnInit, AfterViewChecked {

  @Input() filters: object;
  formFilter: FormGroup;
  @Output() filtersModelchanges: EventEmitter<object> = new EventEmitter<object>();
  submitted: boolean = false;
  filtersVisibility: object = {};
  submitText: string = 'Aplicar';

  constructor(private formBuilder: FormBuilder, private cdRef: ChangeDetectorRef, private formService: FormsValidatorService) {}

  ngOnInit() {
    this.filtersVisibility = this.setFiltersVisibility();
    //this.formFilter = this.formBuilder.group(this.formValidator.generateForm(this.filters,this.filtersVisibility));
    this.formFilter = this.formBuilder.group(this.formService.generateForm(this.filters, this.filtersVisibility));
    // console.log('form reactive', this.formFilter);
    
    /*
    this.formFilter.valueChanges.subscribe( value => {
      console.log('form state', this.formFilter, this.formFilter.getRawValue());
    })
    */

    // ver como agregar los observer en una sola recorrida del array
    for (const filterName of Object.keys(this.filters)) {
      if(this.filters.hasOwnProperty(filterName)){
        const filter = this.filters[filterName];

        if(filter.actions){
          const conditionValues = Object.keys(filter.actions);

          this.formFilter.get(filterName).valueChanges.subscribe( 
            value => {
              if(value === 'PDF'){
                this.submitText = 'Aplicar';
              }else if(value === 'EXCEL'){
                this.submitText = 'Exportar';
              }

              if(conditionValues.includes(value)){
                // si es hidden --> hay que ejecutar accion
                if(filter.actions[value].hidden){
                  // console.log(`value change ${filterName} = ${value}`, filter.actions[value].hidden);
                  this.hideFilterControls(filter.actions[value].hidden);

                }
              }

              this.formFilter = this.formService.updateFormValidators(this.formFilter, this.filters, this.filtersVisibility);
            }
          )
        }

      }
    }

  }

  ngAfterViewChecked(){
    this.cdRef.detectChanges();
  }

  get formControls(){
    return this.formFilter.controls;
  }

  get filtersKey(){
    return Object.keys(this.filters);
  }

  public applyFilters() {
    const oFilter = this.formFilter.getRawValue();
    const data = {};
    // solo devolver datos de campos visibles
    for(let fieldName in this.filtersVisibility){
      if(!this.filtersVisibility[fieldName]){
        data[fieldName] = oFilter[fieldName];
      }
    }
    this.filtersModelchanges.emit(data);
  }
  
  /**  Form Validations **/
  // Luego crear un formService con :
  /*
      1- creacion de formularios por json
      2- validacion de formgroups por tipo de campo
  */

  validateStartDate(control: FormControl){
    if(this.formControls.startDate.value > this.formControls.endDate.value){
      return { endDateLower: true }
    }
    return null;
  }

  validateRangeDate(group: FormGroup) {

    const startDate = group.controls.desde.value;
    const endDate = group.controls.hasta.value;
    
    if( startDate > endDate ) {
      return { greaterThan: true };
    }

    if( startDate < endDate ) {
      return null;
    }

    return null;
  }

  setFiltersVisibility(){
    var visibility = {};
    var fields = Object.keys(this.filters);

    for(let filterName of fields){
      if(this.filters.hasOwnProperty(filterName)){
        if(this.filters[filterName].hidden){
          visibility[filterName] = true;
        }else{
          visibility[filterName] = false;
        }
      }

    }
    return visibility;
  }


  hideFilterControls(visibility){

    for(let filter in visibility){

      if(this.filtersVisibility.hasOwnProperty(filter)){
        this.filtersVisibility[filter] = visibility[filter];
      }

    }
  }

}
