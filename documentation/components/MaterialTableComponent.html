import { Component, OnInit, Input, Output, AfterViewInit,ViewChild, OnChanges } from '@angular/core';
import {SelectionModel} from '@angular/cdk/collections';

import { MatTableDataSource } from '@angular/material/table';
import {MatPaginator} from '@angular/material/paginator';
import {MatSort} from '@angular/material/sort';
import { ToasterService } from 'src/app/core/services/toaster.service';

@Component({
  selector: 'app-material-table',
  templateUrl: './material-table.component.html',
  styleUrls: ['./material-table.component.scss']
})
export class MaterialTableComponent implements OnInit, OnChanges {

  @Input() title: string;
  @Input() dataSource: MatTableDataSource<object[]>;
  @Input() columns: {};
  @Input() details: any[];
  @Input() filtersData?: object;
  @Input() columnsVisibility: object;
  @Input() isLoading: boolean;
  emptyTableText: string = 'No se encontraron registros para mostrar';
  searchText: string;
  sortIsDesc: boolean;
  sortColumn: string;
  sortDirection: number;
  searchFocused: boolean = false;
  // pagination
  itemsPerPage: number;

  // displayedColumns = [];
  selection = new SelectionModel<any>(true, []);

  @ViewChild(MatPaginator, {static: true}) paginator: MatPaginator;
  @ViewChild(MatSort, {static: true}) sort: MatSort;

  constructor(private toaster: ToasterService) {
    this.searchText = '';
  }

  ngOnInit() {
    //this.initColumnsVisibility();

    this.dataSource.paginator = this.paginator;
    this.dataSource.sort = this.sort;
  }

  get displayedColumns(){
    return Object.keys(this.columns);
  }
  
  get itemsQty(): number {
    if(!this.dataSource.data){
      return 0;
    }
    return this.dataSource.data.length;
  }

  get rowsSelectedQty(): number{
    return this.selection.selected.length;
  }

  get tableIsEmpty(){
    return this.itemsQty === 0;
  }
  ngAfterViewInit() {
  }



  applyFilter() {
    this.dataSource.filter = this.searchText.trim().toLowerCase();

    if (this.dataSource.paginator) {
      this.dataSource.paginator.firstPage();
    }
  }

  /** Whether the number of selected elements matches the total number of rows. */
  isAllSelected() {
    const numSelected = this.selection.selected.length;
    if(numSelected === 0){
      return false;
    }
    let pageSize = this.dataSource.paginator.pageSize;
    const lastPageSize = this.dataSource.paginator.length % this.dataSource.paginator.pageSize;
    
    if(!this.dataSource.paginator.hasNextPage() && lastPageSize !== 0){
      pageSize = lastPageSize;
    } 
    return numSelected === pageSize;
  }

  /** Selects all rows if they are not all selected; otherwise clear selection. */
  masterToggle() {
    this.isAllSelected() ? 
    this.selection.clear() : this.selectRows();
  }

  selectRows() {
    let currentPage = this.dataSource.paginator.pageIndex;
    let pageSize = this.dataSource.paginator.pageSize;
    let firstElement =  pageSize * currentPage;
    let lastElement = firstElement + pageSize;

    for (let index = firstElement; index < lastElement; index++) {
      if(this.dataSource.data[index]){
        this.selection.select(this.dataSource.data[index]);
      }
    }
  }

  /** The label for the checkbox on the passed row */
  checkboxLabel(row?: any): string {
    if (!row) {
      return `${this.isAllSelected() ? 'select' : 'deselect'} all`;
    }
    return `${this.selection.isSelected(row) ? 'deselect' : 'select'} row ${row.position}`;
  }

  public downloadSelected(){
    let link = document.createElement('a');
    link.setAttribute('type', 'hidden');
    link.href = 'assets/archivos.zip';
    link.download = './';
    document.body.appendChild(link);
    link.click();
    link.remove();
    this.toaster.show(`Descargando ${this.rowsSelectedQty} elementos`,'');
  }

  public download(){
    let link = document.createElement('a');
    link.setAttribute('type', 'hidden');
    link.href = 'assets/sample.pdf';
    link.download = './';
    document.body.appendChild(link);
    link.click();
    link.remove();
    this.toaster.show('Descargando ...','');
  }

  ngOnChanges(){
    this.dataSource.paginator = this.paginator;
    this.dataSource.sort = this.sort;
  }
  
  clearSearch(event){
    this.searchText = '';
    this.applyFilter();
  }

}
