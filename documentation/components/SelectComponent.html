import { Component, OnInit, Input } from '@angular/core';
import { FormControl } from '@angular/forms';
import { ApiService } from 'src/app/core/services/api.service';


export interface LabelOptions {
  text: string;
  class: string;
}

export interface ControlOptions {
  type: string;
  class: string;
  path: string;
  label: LabelOptions;
  control: object;
}

@Component({
  selector: 'app-select',
  templateUrl: './select.component.html',
  styleUrls: ['./select.component.scss']
})
export class SelectComponent implements OnInit {

  @Input() controlOptions: ControlOptions; 
  @Input() iFormControl: FormControl;
  options: object[];
  
  constructor(private apiService: ApiService) { }

  ngOnInit() {
    if(this.controlOptions.control['path']){
      this.apiService.get(this.controlOptions.control['path']).subscribe( (data: object[]) =>{
        this.options = data;
      });
    }else{
      this.options = this.controlOptions.control['options'];
    }
  }

  get controlSpecs(){
    return this.controlOptions.control;
  }

  get label(){
    return this.controlOptions.label.text || this.controlOptions.label;
  }

  change(event){
    console.log('change combobox', event.value);
  }

}

<mat-form-field class="w-100" appearance="outline">
  <mat-label>{{ label }}</mat-label>
  <mat-select 
  [formControl]="iFormControl"
  [multiple]="controlSpecs['multiselect']"
  (selectionChange)="change($event)">
    <mat-option *ngIf="controlSpecs['hasEmptyOption']"></mat-option>
    <mat-option *ngFor="let option of options" [value]="option[controlSpecs['pasteFieldOnSelect']] || option['value']">
      {{ option[controlSpecs['options']['value']] || option.value }}
    </mat-option>
  </mat-select>
  <mat-error *ngIf="iFormControl.errors?.required">Por favor complete este campo</mat-error>
</mat-form-field>
