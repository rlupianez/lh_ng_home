@Component({
  selector: 'app-cuenta-corriente-list',
  templateUrl: './cuenta-corriente-list.component.html',
  styleUrls: ['./cuenta-corriente-list.component.scss']
})
export class CuentaCorrienteListComponent implements OnInit {

  tableTitle: string = 'Cuenta Corriente'
  panelExpanded: boolean = true;
  data: any = [];
  columns: object;
  rowDetails: any[];
  filterModel: object;
  columnsVisibility: object;
  isLoadingResults: boolean = false;
  filtersFields: object = CtaCteFilters;

  constructor(private apiService: ApiService, private router: Router, private toaster: ToasterService) { }

  ngOnInit() {

    this.columns = {
      select : { label: 'Select', class: 'col' },
      productor : { label: 'Productor', visible: true, class: 'col' },
      periodo: { label: 'Periodo', visible: true, class: 'col' },
      formato: { label: 'Formato', visible: true, class: 'col' },
      descargar: { label: 'Descargar', class: 'col', action: true, icon: 'far fa-file-pdf fa-lg' }
    }
  }

  applyFilters(oFilterData: object) {
  
    if( oFilterData['tipo'] === "PDF"){
      delete oFilterData['tipo'];
      const filterString = this.apiService.generateStringFilter(oFilterData, this.filtersFields);
    
      this.isLoadingResults = true;
      this.apiService.get(`/cuenta-corriente-pdf${filterString}`).subscribe( (data : object[]) => {
        console.log('cuenta corriente get', data);
        this.data = new MatTableDataSource(data);
        delay(5000);
        this.isLoadingResults = false;
        if(data.length > 0){
          this.panelExpanded = false;
        }
      });
    }else{
      let link = document.createElement('a');
      link.setAttribute('type', 'hidden');
      link.href = 'assets/sample_excel.xlsx';
      link.download = './';
      document.body.appendChild(link);
      link.click();
      link.remove();
      this.toaster.show('Descargando ...','');
    }
    
  }

}
