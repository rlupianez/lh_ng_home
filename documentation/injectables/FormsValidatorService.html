import { Injectable } from '@angular/core';
import { FormControl, FormGroup, AbstractControl, Validators, FormBuilder } from '@angular/forms';

import * as moment from 'moment';

@Injectable({
  providedIn: 'root'
})
export class FormsValidatorService {

  validatorsTypes: object = {
    'monthyear-range': [this.validateRangeDate],
    'date-range': [this.validateRangeDate],
    'typeahead': [this.requireMatchOption.bind(this)],
    'combobox': [],
    'date': []
  }
  

  constructor(private formBuilder: FormBuilder) { }

  formControlIsRequired(control: FormControl){

  }

  getControlValidators(controlType: string, required: boolean){
    if(!controlType){
      return null
    }

    let validators = this.validatorsTypes[controlType];

    // si el tipo es range, es un group y el required va en los controls y no en el formgroup
    if(required && !['date-range','monthyear-range'].includes(controlType)){
      validators.push(Validators.required);
    }

    return validators;
  }
  
  validateRangeDate(group: FormGroup) {

    const startDate = group.controls.fechaDesde.value;
    const endDate = group.controls.fechaHasta.value;
    
    if( startDate > endDate ) {
      return { greaterThan: true };
    }

    if( startDate < endDate ) {
      return null;
    }

    return null;
  }

  requireMatchOption(control: AbstractControl) {
    const selection: any = control.value;
    if (typeof selection === 'string') {
        return { incorrect: true };
    }
    return null;
  }

  /*
    Genera objeto formularo en base a objeto especificado de opciones
  */

  public generateForm(filters: object, filtersVisibility: object){
    let form = {};
    

    for (const filterName of Object.keys(filters)) {
      let validators = [];
      if(filters.hasOwnProperty(filterName)){

        const filter = filters[filterName];
        // this.getFieldValidator(filterName, filter);
        if(filter.required && !filtersVisibility[filterName]){
          validators.push( Validators.required );
        }
        
        switch(filter.control.type){
          case 'monthyear-range':
          case 'date-range':
              form[filterName] = this.formBuilder.group({
                fechaDesde: [moment('01/01/2010','DD/MM/YYYY'), validators],
                fechaHasta: ['', validators],
              }, { validator: this.validateRangeDate });


              break;
          case 'typeahead':

              if(filter.required){
                  //validators.push(RequireMatch.bind(this));
                  form[filterName] = new FormControl('', validators);   
              }else {
                form[filterName] = new FormControl('');   
              }

              break;
          case 'datepicker':

                if(filter.required){
                  form[filterName] = new FormControl(moment(new Date(),'DD/MM/YYYY'), validators);   
                }else {
                  form[filterName] = new FormControl(moment(new Date(),'DD/MM/YYYY'), );  
                }
  
                break;
          default:
              
              form[filterName] = new FormControl('', validators);
              break;
        }

      }
      

    }

    return form;
  }

  /*
    Actualiza los validators de los controles dinamicos del formulario
    segun su visibilidad, si estan visibles o no
  */
  updateFormValidators(formFilter: FormGroup, filters: object, filtersVisibility: object){
    for(let filter in filtersVisibility){

      if(!filters[filter].hidden){
        continue;
      }

      if(!filters[filter].required){
        continue;
      }

      if(filtersVisibility[filter]){

        if(filters[filter].control.type === 'date-range' || filters[filter].control.type === 'monthyear-range'){
          let group = formFilter.get(filter);
          group.get('fechaDesde').clearValidators();
          group.get('fechaHasta').clearValidators();
          group.clearValidators();
        }else{
          formFilter.get(filter).clearValidators();
        }
      }else {
        if(filters[filter].control.type === 'date-range' || filters[filter].control.type === 'monthyear-range'){
          let group = formFilter.get(filter);
          group.get('fechaDesde').setValidators([Validators.required]);
          group.get('fechaHasta').setValidators([Validators.required]);
          group.setValidators(this.validateRangeDate);
        }else{
          formFilter.get(filter).setValidators([Validators.required]);  
        }
        
      }

      if(filters[filter].control.type === 'date-range' || filters[filter].control.type === 'monthyear-range'){
        let group = formFilter.get(filter);
          group.updateValueAndValidity();
          group.get('fechaDesde').updateValueAndValidity();
          group.get('fechaHasta').updateValueAndValidity();
      }else{
        formFilter.get(filter).updateValueAndValidity();  
      }
      

    }

    return formFilter;
  }

  getControlValidator(type: string, controlOptions: object){

  }
}
